$realValue:=0
C_LONGINT($foundRow)

C_POINTER($objPtr)
$objPtr:=OBJECT Get pointer(Object named;$inListBox)

  //リストボックスの最初に見つかった選択されている行番号を得る
$foundRow:=JCL_lst_Selected_firstRow ($objPtr)
If ($foundRow#0)

	$realValue:=$inDataAryPtr->{$foundRow}
	
End if
