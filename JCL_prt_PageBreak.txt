  //印刷対象が複数ページにまたがるときのページブレーク

C_LONGINT($1;$nombre)
$nombre:=$1  //ページ番号
C_LONGINT($2;$numOfPages)
$numOfPages:=$2  //総ページ数

If ($nombre<$numOfPages)
	
	PAGE BREAK(>)  //途中のページを印刷
	
Else 
	
	If ($nombre=$numOfPages)
		
		PAGE BREAK  //最後のページを印刷
		
	End if 
	
End if
