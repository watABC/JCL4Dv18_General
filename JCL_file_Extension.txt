  //ファイル名から拡張子を取得

C_TEXT($1;$fileName)
$fileName:=$1

C_TEXT($0;$ext)

If ($fileName="")
	$ext:=""
Else 
	
	ARRAY TEXT($aryExtract;0)
	$sizeOfAry:=JCL_str_Extract ($fileName;".";->$aryExtract)
	
	If ($sizeOfAry=1)
		
		$ext:=""
		
	Else 
		
		$ext:=$aryExtract{$sizeOfAry}
		
	End if 
	
End if 

$0:=$ext
