  //リストボックスで1行以上選択されていればボタンはイネイブル

C_POINTER($1;$inListBoxPtr)
$inListBoxPtr:=$1  //選択判定対象となるリストボックスのブーリアン配列
C_POINTER($2;$inButtonPtr)
$inButtonPtr:=$2  //イネイブル対象となるボタンのポインタ
C_LONGINT($selCnt)

  //リストで選択されている行が一つ以上の場合にイネイブル、
  //そうでなければボタンはグレイ
$selCnt:=JCL_lst_SelectedCount ($inListBoxPtr)
If ($selCnt>=1)
	
	JCL_btn_SetEnable ($inButtonPtr;True)
	
Else 
	
	JCL_btn_SetEnable ($inButtonPtr;False)
	
End if
