  //携帯番号をフォーマット（ハイフンを入れる）

C_TEXT($1;$keitai)
$keitai:=$1

C_TEXT($0;$formatKeitai)
$formatKeitai:=""

C_TEXT($part1;$part2;$part3)

If (Length($keitai)#0)
	
	$part1:=Substring($keitai;1;3)
	$part2:=Substring($keitai;4;4)
	$part3:=Substring($keitai;8)
	
	$formatKeitai:=$part1+"-"+$part2+"-"+$part3
	
End if 

$0:=$formatKeitai
