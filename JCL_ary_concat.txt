  // 配列と配列を　連結して　配列に追加して返す　親でアロケート

C_POINTER($1;$aryResultPtr)
$aryResultPtr:=$1  // 結果配列、渡された時に要素があればそのあとに追加
C_POINTER($2;$aryPtr)
$aryPtr:=$2  // 追加する配列
C_LONGINT($0;$cnt)
$cnt:=0
C_LONGINT($i;$sizeOfAry)

$sizeOfAry:=Size of array($aryPtr->)
For ($i;1;$sizeOfAry)
	
	APPEND TO ARRAY($aryResultPtr->;$aryPtr->{$i})
	$cnt:=$cnt+1
	
End for 

$0:=$sizeOfAry
