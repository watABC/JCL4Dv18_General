  //ログをファイルに書き出す
  //ファイルはデスクトップ、なければ作る

C_TEXT($1;$inMsg)
$inMsg:=$1  //書き出したい内容
C_TEXT($folderPath;$filePath)
C_TIME($FileRef)  //ファイル参照

$folderPath:=System folder(Desktop)
$filePath:=JCL_file_MakeFilePath ($folderPath;"Logout.txt")

$open_ok:=JCL_file_OpenForWrite ($filePath;->$FileRef)
If ($open_ok=1)
	
	  //ファイルに書き出す
	$dateStr:=JCL_str_Datemark (Current date;Current time)
	JCL_file_WriteSJIS ($FileRef;$dateStr)
	JCL_file_WriteTab ($FileRef)
	JCL_file_WriteSJIS ($FileRef;$inMsg)
	JCL_file_WriteCRLF ($FileRef)
	
	  //ファイルを閉じる
	CLOSE DOCUMENT($FileRef)
	
End if
