  //リストボックスを行番号で選択
  //行番号がゼロの場合は何も選択しない、という仕様にした。

C_POINTER($1;$inListBoxPtr)
$inListBoxPtr:=$1  //ターゲットとなるリストボックスのブーリアン配列
C_LONGINT($2;$row_number)
$row_number:=$2  //行番号

  //行番号がゼロの場合は何も選択しない
If ($row_number#0)
	LISTBOX SELECT ROW($inListBoxPtr->;$row_number;lk replace selection)
	
Else 
	  //ゼロの場合はデセレクト
	LISTBOX SELECT ROW($inListBoxPtr->;0;lk remove from selection)
	
End if
