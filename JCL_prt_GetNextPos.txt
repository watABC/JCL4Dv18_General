  //現在の印刷位置から、次のセクションが印刷可能領域に収まるかを判断、収まらなければゼロを返す
  //20171029 wat

C_LONGINT($1;$inPos)
$inPos:=$1
C_LONGINT($2;$section_height)
$section_height:=$2
C_LONGINT($0;$result_feed_pos)
C_LONGINT($page_height)

GET PRINTABLE AREA($page_height)
If (($inPos+$section_height)<$page_height)
	
	  //ページに残りの印刷領域がある場合は　続けて印刷
	$result_feed_pos:=$inPos+$section_height
	
Else 
	
	  //ページに残りの印刷領域がない場合は　ゼロを返す
	$result_feed_pos:=0
	
End if 

$0:=$result_feed_pos
