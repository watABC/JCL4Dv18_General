C_TEXT($fileName)

$fileName:="SQL"+JCL_str_Datemark (Current date;Current time)+".txt"
$folderPath:=System folder(Desktop)

If ($open_ok=1)

	JCL_file_WriteSJIS ($FileRef;$inMsg)

	CLOSE DOCUMENT($FileRef)
	
End if
