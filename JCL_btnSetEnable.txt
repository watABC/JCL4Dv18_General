  //ボタンをイネイブルにする、名前で指定する
  //引数のブール値が、真だったらイネイブル

C_TEXT($1;$inBtnName)
$inBtnName:=$1  //イネイブル対象となるボタンのポインタ
C_BOOLEAN($inBool)
$inBool:=$2

  //引数が真ならボタンはイネイブル、偽ならグレイ
If ($inBool=True)
	OBJECT SET ENABLED(*;$inBtnName;True)
	
Else 
	OBJECT SET ENABLED(*;$inBtnName;False)
	
End if
