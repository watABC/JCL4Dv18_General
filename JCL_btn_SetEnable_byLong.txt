  //ボタンをイネイブルにする、ラッパ

C_POINTER($1;$inButtonPtr)
$inButtonPtr:=$1  //イネイブル対象となるボタンのポインタ
C_LONGINT($2;$inLong)
$inLong:=$2

  //引数が１以上ならボタンはイネイブル、ゼロ以下ならグレイ
If ($inLong>=1)
	JCL_btn_SetEnable ($inButtonPtr;True)
Else 
	JCL_btn_SetEnable ($inButtonPtr;False)
End if
